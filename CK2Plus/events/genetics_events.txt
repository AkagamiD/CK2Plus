#Reduce fertility / change stats for older characters
character_event = {
	id = 300003
	picture = "GFX_evt_illness"

	is_triggered_only = yes
	hide_from = yes

	trigger = {
		ai = yes
		is_ruler = no
		liege = {
			NOT = { trait = immortal }
			age = 50
			NOT = { age = 60 }
			ai = no
			NOT = { trait = old50 }
		}
	}

	mean_time_to_happen = { days = 1 }

	desc = "AI_EVENT"

	option = {
		name = "OK"
		liege = {
			add_trait = old50
			remove_trait = young1
			remove_trait = young6
			remove_trait = young16
		}
	}
}

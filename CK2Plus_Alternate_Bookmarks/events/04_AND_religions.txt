### Prophets appear
#Liege conversion in response to courtier/vassal conversion
character_event = {
	id = AND.1
	desc = AND.1.desc
	picture = "GFX_evt_council"
	is_triggered_only = yes
	
	option = {
		name = AND.1.a
		ROOT = {
			religion = FROM
		}
		FROM = {
			IF = {
				limit = {
					NOT = {
						has_job_title = job_spiritual
					}
				}
				remove_title = job_chancellor
				remove_title = job_marshal
				remove_title = job_steward
				remove_title = job_spymaster
				give_job_title = job_spiritual
			}
		}
		any_realm_province = {
			limit = {
				any_province_character = {
					AND = {
						has_character_modifier = prophet
						religion = FROM
					}
				}
			}
			religion = FROM
			any_province_character = { religion = FROM }
		}
		IF = {
			limit = {
				independent = no
			}
			liege = {
				character_event = { id = AND.2 }
			}
		}
	}
	option = {
		name = AND.1.b
		trigger = {
			ai = no
		}
		FROM = {
			banish = yes
			hidden_tooltip = {
				character_event = { id = AND.3 }
			}
		}
	}
}

# neighbouring province conversion
province_event = {
	id = AND.4
	desc = AND.4.desc
	picture = "GFX_evt_council"
	
	trigger = {
		religion = ancestry
		any_neighbor_province = { # Only when the neighbouring province has a new religion and within the same realm
			has_owner = yes
			owner = {
				ROOT = {
					owner = {
						same_realm = PREVPREV
					}
				}
			}
			NOT = {
				religion = ancestry
			}
		}
	}
	
	mean_time_to_happen = {
		days = 3650
	}
	
	immediate = {
		random_neighbor_province = {
			limit = {
				NOT = { religion = ancestry }
				owner = { # which is within the same realm
					ROOT = {
						owner = {
							same_realm = PREVPREV
						}
					}
				}
			}
			reverse_religion = ROOT
		}
	}
	
	option = {
		name = AND.4.a
		if = {
			limit = { 
				ruler = { religion = ancestry }
			}
			ruler = {
				religion = ROOT
			}
		}
	}
}

### Relative conversion events
# Liege conversion in response to courtier/vassal conversion
character_event = {
	id = 1000190001
	desc = EVTDESC1000190001
	picture = "GFX_evt_council"
	is_triggered_only = yes
	
	option = {
		name = EVTOPTA1000190001
		ROOT = {
			religion = FROM
		}
		FROM = {
			IF = {
				limit = {
					NOT = {
						has_job_title = job_spiritual
					}
				}
				remove_title = job_chancellor
				remove_title = job_marshal
				remove_title = job_steward
				remove_title = job_spymaster
				give_job_title = job_spiritual
			}
		}
		any_realm_province = {
			limit = {
				any_province_character = {
					AND = {
						has_character_modifier = prophet
						religion = FROM
					}
				}
			}
			religion = FROM
			any_province_character = { religion = FROM }
		}
		IF = {
			limit = {
				independent = no
			}
			liege = {
				character_event = { id = 1000190002 }
			}
		}
	}
	option = {
		name = EVTOPTB1000190001
		trigger = {
			ai = no
		}
		FROM = {
			banish = yes
			hidden_tooltip = {
				character_event = { id = 1000190003 }
			}
		}
	}
}

# Religion spread event
province_event = {
	id = 1000190004
	desc = EVTDESC1000190004
	picture = "GFX_evt_council"
	
	trigger = {
		religion = folklore
		any_neighbor_province = { # Only when the neighbouring province has a new religion and within the same realm
			has_owner = yes
			owner = {
				ROOT = {
					owner = {
						same_realm = PREVPREV
					}
				}
			}
			NOT = {
				religion = folklore
			}
		}
	}
	
	mean_time_to_happen = {
		days = 3650
	}
	
	immediate = {
		random_neighbor_province = {
			limit = {
				NOT = { religion = folklore }
				owner = { # which is within the same realm
					ROOT = {
						owner = {
							same_realm = PREVPREV
						}
					}
				}
			}
			reverse_religion = ROOT
		}
	}
	
	option = {
		name = EVTOPTA1000190004
		if = {
			limit = { 
				ruler = { religion = folklore }
			}
			ruler = {
				religion = ROOT
			}
		}
	}
}

	id = 1000190009
	hide_window = yes
	hide_from = yes
	
	trigger = {
		has_province_flag = prophet
	}
	
	immediate = {
		clr_province_flag = prophet
		IF = {
			limit = {
				has_global_flag = ancestral_founded

			clr_global_flag = astrolatry_founded

		IF = {
			limit = {
				NOT = {
					any_realm_character = {
						has_character_modifier = prophet
					}
				}
			}
			random_list = {	# random allocation of religions. The follow-up events can result in failure if the religion already has been founded

					random_province_character = {
						limit = {
							ai = yes

								trait = detached_priest

								trait = mastermind_theologian
								trait = scholar
							}
						}
						character_event = { id = 1000190020 }
					}
